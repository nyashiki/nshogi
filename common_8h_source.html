23template <core::Color C, bool ChannelsFirst = true>
24inline std::size_t getMoveIndex(core::Move16 M16) {
25 const core::Square To =
26 (C == core::Black) ? M16.to() : core::getInversed(M16.to());
27
28 if (M16.drop()) {
29 const auto Type =
30 (core::PieceTypeKind)(M16.from() - core::NumSquares + 1);
31
32 if constexpr (ChannelsFirst) {
33 return (std::size_t)(20 + (int)Type - 1) *
34 (std::size_t)core::NumSquares +
35 (std::size_t)To;
36 } else {
37 return (std::size_t)To * 27 + (std::size_t)(20 + (int)Type - 1);
38 }
39 }
40
41 const core::Square From =
42 (C == core::Black) ? M16.from() : core::getInversed(M16.from());
43 const uint8_t SerializedDirection =
44 core::SquareSerializedDirection[(std::size_t)From][(std::size_t)To];
45
46 const std::size_t Channel =
47 (M16.promote()) ? (10 + SerializedDirection) : SerializedDirection;
48
49 if constexpr (ChannelsFirst) {
50 return Channel * (std::size_t)core::NumSquares + (std::size_t)To;
51 } else {
52 return (std::size_t)To * 27 + Channel;
53 }
54}
