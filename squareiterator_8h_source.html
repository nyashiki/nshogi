22enum struct IterateOrder {
23 ESWN, // from eastsouth to westnorth (Sq1I, Sq1H, ..., Sq1A, Sq2I, Sq2H, ..., Sq9A).
24 WNES, // from westnorth to eastsouth (Sq9A, Sq9B, ..., Sq9I, Sq8A, Sq8B, ..., Sq1I).
25 NWSE, // from northwest to southeast (Sq9A, Sq8A, ..., Sq1A, Sq9B, Sq8B, ..., Sq1I).
26 SENW, // from southeast to northwest (Sq1I, Sq2I, ..., Sq9I, Sq1H, Sq2H, ..., Sq9A).
27 Fastest = ESWN,
28};

57template <IterateOrder Order>
58struct SquareIterator {
59 public:
60 SquareIterator() {
61 if constexpr (Order == IterateOrder::Fastest ||
62 Order == IterateOrder::ESWN) {
63 Sq = (Square)0;
64 } else if constexpr (Order == IterateOrder::NWSE) {
65 Sq = Sq9A;
66 R = RankA;
67 F = File9;
68 } else if constexpr (Order == IterateOrder::SENW) {
69 Sq = Sq1I;
70 R = RankI;
71 F = File1;
72 } else if constexpr (Order == IterateOrder::WNES) {
73 Sq = Sq9A;
74 }
75 }

100 SquareIterator<Order>& operator++() {
101 if constexpr (Order == IterateOrder::Fastest ||
102 Order == IterateOrder::ESWN) {
103 Sq = (Square)(Sq + 1);
104 return *this;
105 } else if constexpr (Order == IterateOrder::WNES) {
106 if (Sq == Sq1I) {
107 Sq = NumSquares;
108 return *this;
109 }
110
111 Sq = (Square)(Sq - 1);
112 return *this;
113 } else if constexpr (Order == IterateOrder::NWSE) {
114 if (F == File1) {
115 if (R == RankI) {
116 Sq = NumSquares;
117 return *this;
118 }
119
120 R = (Rank)(R - 1);
121 F = File9;
122 } else {
123 F = (File)(F - 1);
124 }
125 } else if constexpr (Order == IterateOrder::SENW) {
126 if (F == File9) {
127 if (R == RankA) {
128 Sq = NumSquares;
129 return *this;
130 }
131
132 R = (Rank)(R + 1);
133 F = File1;
134 } else {
135 F = (File)(F + 1);
136 }
137 }
138
139 Sq = makeSquare(R, F);
140 return *this;
141 }
142
143 SquareIterator<Order>& operator+=(int Inc) {
144 for (int I = 0; I < Inc; ++I) {
145 ++(*this);
146 }
147
148 return *this;
149 }

151 private:
152 Square Sq;
153
154 Rank R;
155 File F;
156};
