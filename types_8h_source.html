275static constexpr auto DirectionDataInternal =
276 []() -> std::pair<std::array<std::array<Direction, NumSquares>, NumSquares>,
277 std::array<std::array<uint8_t, NumSquares>, NumSquares>> {
278 std::array<std::array<Direction, NumSquares>, NumSquares> Directions = {};
279 std::array<std::array<uint8_t, NumSquares>, NumSquares>
280 SerializedDirections = {};
281
282 for (Square Sq1 : Squares) {
283 for (Square Sq2 : Squares) {
284 if (Sq1 == Sq2) {
285 Directions[(std::size_t)Sq1][(std::size_t)Sq2] = (Direction)0;
286 }
287
288 const Rank R1 = squareToRank(Sq1);
289 const Rank R2 = squareToRank(Sq2);
290 const File F1 = squareToFile(Sq1);
291 const File F2 = squareToFile(Sq2);
292 const bool Comp = (int)Sq1 > (int)Sq2;
293
294 if (R1 == R2) {
295 Directions[(std::size_t)Sq1][(std::size_t)Sq2] =
296 Comp ? East : West;
297 SerializedDirections[(std::size_t)Sq1][(std::size_t)Sq2] =
298 Comp ? 2 : 6;
299 } else if (F1 == F2) {
300 Directions[(std::size_t)Sq1][(std::size_t)Sq2] =
301 Comp ? South : North;
302 SerializedDirections[(std::size_t)Sq1][(std::size_t)Sq2] =
303 Comp ? 4 : 0;
304 } else if (Sq2 == Sq1 + NorthNorthEast && F1 != File1) {
305 Directions[(std::size_t)Sq1][(std::size_t)Sq2] = NorthNorthEast;
306 SerializedDirections[(std::size_t)Sq1][(std::size_t)Sq2] = 8;
307 } else if (Sq2 == Sq1 + NorthNorthWest && F1 != File9) {
308 Directions[(std::size_t)Sq1][(std::size_t)Sq2] = NorthNorthWest;
309 SerializedDirections[(std::size_t)Sq1][(std::size_t)Sq2] = 9;
310 } else if (Sq2 == Sq1 + SouthSouthEast && F1 != File1) {
311 Directions[(std::size_t)Sq1][(std::size_t)Sq2] = SouthSouthEast;
312 SerializedDirections[(std::size_t)Sq1][(std::size_t)Sq2] = 10;
313 } else if (Sq2 == Sq1 + SouthSouthWest && F1 != File9) {
314 Directions[(std::size_t)Sq1][(std::size_t)Sq2] = SouthSouthWest;
315 SerializedDirections[(std::size_t)Sq1][(std::size_t)Sq2] = 11;
316 } else if ((int)F1 - (int)F2 == (int)R1 - (int)R2) {
317 Directions[(std::size_t)Sq1][(std::size_t)Sq2] =
318 Comp ? SouthEast : NorthWest;
319 SerializedDirections[(std::size_t)Sq1][(std::size_t)Sq2] =
320 Comp ? 3 : 7;
321 } else if ((int)F1 - (int)F2 == (int)R2 - (int)R1) {
322 Directions[(std::size_t)Sq1][(std::size_t)Sq2] =
323 Comp ? NorthEast : SouthWest;
324 SerializedDirections[(std::size_t)Sq1][(std::size_t)Sq2] =
325 Comp ? 1 : 5;
326 }
327 }
328 }
329
330 return {Directions, SerializedDirections};
331}();

364static constexpr uint32_t StandMasks[NumStandKinds] = {
365 0U,
366 0b11111U << StandShifts[PTK_Pawn ],
367 0b00111U << StandShifts[PTK_Lance ],
368 0b00111U << StandShifts[PTK_Knight],
369 0b00111U << StandShifts[PTK_Silver],
370 0b00011U << StandShifts[PTK_Bishop],
371 0b00011U << StandShifts[PTK_Rook ],
372 0b00111U << StandShifts[PTK_Gold ],

383}

386inline constexpr Stands incrementStand(Stands St, PieceTypeKind Pt) {
387 return (Stands)(St + StandBits[Pt]);

396[[maybe_unused]]
397inline constexpr Stands constructStand(uint8_t PawnCount, uint8_t LanceCount,
398 uint8_t KnightCount, uint8_t SilverCount,

401 return (Stands)((uint32_t)PawnCount * StandBits[PTK_Pawn] +
402 (uint32_t)LanceCount * StandBits[PTK_Lance] +

404 (uint32_t)SilverCount * StandBits[PTK_Silver] +
405 (uint32_t)BishopCount * StandBits[PTK_Bishop] +
406 (uint32_t)RookCount * StandBits[PTK_Rook] +
407 (uint32_t)GoldCount * StandBits[PTK_Gold]);

423inline constexpr bool isSuperiorOrEqual(Stands St1, Stands St2) {
424 constexpr uint32_t Mask =
425 1U << (StandShifts[PTK_Pawn] + 5) | 1U << (StandShifts[PTK_Lance] + 3) |
426 1U << (StandShifts[PTK_Knight] + 3) |
427 1U << (StandShifts[PTK_Silver] + 3) |
428 1U << (StandShifts[PTK_Bishop] + 2) |
429 1U << (StandShifts[PTK_Rook] + 2) | 1U << (StandShifts[PTK_Gold] + 3);
430
431 return ((St1 - St2) & Mask) == 0;
432}
