114 std::vector<float> extract() const {

122 template <core::IterateOrder Order = core::IterateOrder::NWSE,
123 bool ChannelsFirst = true>
124 void extract(float* Dest) const {
125 if constexpr (ChannelsFirst) {
126 for (std::size_t Channel = 0; Channel < sizeof...(FeatureTypes);
127 ++Channel) {
128 Features[Channel].template extract<Order>(
129 Dest + Channel * core::NumSquares);
130 }
131 } else {
132 core::SquareIterator<Order> SquareIt;
133 for (auto It = SquareIt.begin(); It != SquareIt.end(); ++It) {
134 for (std::size_t Channel = 0; Channel < sizeof...(FeatureTypes);
135 ++Channel) {
136 const uint64_t* BB = reinterpret_cast<const uint64_t*>(
137 Features[Channel].data());
138
139 const bool IsRotated = Features[Channel].isRotated();
140
141 if (!IsRotated) {
142 if (*It < 63) {
143 const bool IsSet = (BB[0] & (1ULL << *It)) != 0;
144 Dest[(std::size_t)*It * Features.size() + Channel] =
145 IsSet ? Features[Channel].getValue() : 0.0f;
146 } else {
147 const bool IsSet =
148 (BB[1] & (1ULL << (*It - 63))) != 0;
149 Dest[(std::size_t)*It * Features.size() + Channel] =
150 IsSet ? Features[Channel].getValue() : 0.0f;
151 }
152 } else {
153 if (*It < 63) {
154 const bool IsSet = (BB[0] & (1ULL << *It)) != 0;
155 Dest[(std::size_t)(core::NumSquares - 1 - *It) *
156 Features.size() +
157 Channel] =
158 IsSet ? Features[Channel].getValue() : 0.0f;
159 } else {
160 const bool IsSet =
161 (BB[1] & (1ULL << (*It - 63))) != 0;
162 Dest[(std::size_t)(core::NumSquares - 1 - *It) *
163 Features.size() +
164 Channel] =
165 IsSet ? Features[Channel].getValue() : 0.0f;
166 }
167 }
168 }
169 }
170 }
171 }
