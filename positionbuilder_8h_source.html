60 void setStand(Color C, PieceTypeKind Pt, uint32_t Count) {
61 // The stand representation of `Stands` is kind of
62 // tricky. To set the number of a standing piece `Pt`,
63 // increment/decrement it one by one since this is
64 // not a speed-intended part.
65 while (getStandCount(Instance.getStand(C), Pt) < Count) {
66 Instance.incrementStand(C, Pt);
67 }
68
69 while (getStandCount(Instance.getStand(C), Pt) > Count) {
70 Instance.decrementStand(C, Pt);
71 }
72 }
